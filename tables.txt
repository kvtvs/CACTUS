CREATE TABLE Käyttäjä
(
  KäyttäjäID INT NOT NULL,
  Salasana VARCHAR NOT NULL,
  Käyttäjänimi VARCHAR NOT NULL,
  Sähköposti VARCHAR NOT NULL,
  Rooli INT NOT NULL,
  PRIMARY KEY (KäyttäjäID)
);

CREATE TABLE Tuote
(
  Nimi VARCHAR NOT NULL,
  Kuvaus VARCHAR NOT NULL,
  Hinta FLOAT NOT NULL,
  Julkaisu_pvm DATE NOT NULL,
  TuoteID INT NOT NULL,
  KäyttäjäID INT NOT NULL,
  PRIMARY KEY (TuoteID),
  FOREIGN KEY (KäyttäjäID) REFERENCES Käyttäjä(KäyttäjäID)
);

CREATE TABLE Kuvat
(
  Kuvan_ID INT NOT NULL,
  Kuvan_sijainti VARCHAR NOT NULL,
  TuoteID INT NOT NULL,
  PRIMARY KEY (Kuvan_ID),
  FOREIGN KEY (TuoteID) REFERENCES Tuote(TuoteID)
);

CREATE TABLE tykkäys
(
  KäyttäjäID INT NOT NULL,
  TuoteID INT NOT NULL,
  PRIMARY KEY (KäyttäjäID, TuoteID),
  FOREIGN KEY (KäyttäjäID) REFERENCES Käyttäjä(KäyttäjäID),
  FOREIGN KEY (TuoteID) REFERENCES Tuote(TuoteID)
);
