CREATE TABLE Käyttäjä
(
  KäyttäjäID INT NOT NULL AUTO_INCREMENT,  
  Käyttäjänimi TEXT NOT NULL,
  Sähköposti TEXT NOT NULL,
  Salasana TEXT NOT NULL,
  Rooli INT NOT NULL,
  PRIMARY KEY (KäyttäjäID)
);

CREATE TABLE Tuote
(
  TuoteID INT NOT NULL AUTO_INCREMENT,
  Nimi TEXT NOT NULL,
  Kuvaus TEXT NOT NULL,
  Hinta FLOAT NOT NULL,
  Filename TEXT,
  Julkaisu_pvm DATE NOT NULL CURRENT_TIMESTAMP(),
  KäyttäjäID INT NOT NULL,
  PRIMARY KEY (TuoteID),
  FOREIGN KEY (KäyttäjäID) REFERENCES Käyttäjä(KäyttäjäID)
);

CREATE TABLE Kuvat
(
  Kuvan_ID INT NOT NULL AUTO_INCREMENT,
  Kuvan_sijainti TEXT NOT NULL,
  TuoteID INT NOT NULL,
  PRIMARY KEY (Kuvan_ID),
  FOREIGN KEY (TuoteID) REFERENCES Tuote(TuoteID)
);

CREATE TABLE Tykkäys
(
  KäyttäjäID INT NOT NULL,
  TuoteID INT NOT NULL,
  PRIMARY KEY (KäyttäjäID, TuoteID),
  FOREIGN KEY (KäyttäjäID) REFERENCES Käyttäjä(KäyttäjäID),
  FOREIGN KEY (TuoteID) REFERENCES Tuote(TuoteID)
);
